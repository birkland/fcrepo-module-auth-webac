105     /**
106      *  For a given FedoraResource, get a mapping of acl:agent values to acl:mode values.
107      */
108     private Map<String, List<String>> getAgentRoles(final FedoraResource resource) {
109         LOGGER.debug("Getting agent roles for: {}", resource.getPath());
110 
111         // Get the effective ACL by searching the target node and any ancestors.
112         final Optional<Pair<URI, FedoraResource>> effectiveAcl = getEffectiveAcl(resource);
113 
114         // Construct a list of acceptable acl:accessTo values for the target resource.
115         final List<String> resourcePaths = new ArrayList<>();
116         resourcePaths.add(FEDORA_INTERNAL_PREFIX + resource.getPath());
117         // Construct a list of acceptable acl:accessToClass values for the target resource.
118         final List<URI> rdfTypes = resource.getTypes();
119 
120         // Add the resource location and types of the ACL-bearing parent,
121         // if present and if different than the target resource.
122         effectiveAcl
123             .map(x -> x.getRight())
124             .filter(x -> !x.getPath().equals(resource.getPath()))
125             .ifPresent(x -> {
126                 resourcePaths.add(FEDORA_INTERNAL_PREFIX + x.getPath());
127                 rdfTypes.addAll(x.getTypes());
128             });
129 
130         // Create a function to check acl:accessTo, scoped to the given resourcePaths
131         final Predicate<WebACAuthorization> checkAccessTo = accessTo.apply(resourcePaths);
132 
133         // Create a function to check acl:accessToClass, scoped to the given rdf:type values,
134         // but transform the URIs to Strings first.
135         final Predicate<WebACAuthorization> checkAccessToClass =
136             accessToClass.apply(rdfTypes.stream().map(URI::toString).collect(Collectors.toList()));
137 
138         // Read the effective Acl and return a list of acl:Authorization statements
139         final List<WebACAuthorization> authorizations = effectiveAcl
140                 .map(uncheck(x -> getAuthorizations(x.getLeft().toString())))
141                 .orElse(new ArrayList<>());
142 
143         // Filter the acl:Authorization statements so that they correspond only to statements that apply to
144         // the target (or acl-bearing ancestor) resource path or rdf:type.
145         // Then, assign all acceptable acl:mode values to the relevant acl:agent values: this creates a UNION
146         // of acl:modes for each particular acl:agent.
147         final Map<String, Set<String>> effectiveRoles = new HashMap<>();
148         authorizations.stream()
149             .filter(x -> checkAccessTo.test(x) || checkAccessToClass.test(x))
150             .forEach(x -> {
151                 x.getAgents().stream()
152                     .forEach(y -> {
153                         effectiveRoles.putIfAbsent(y, new HashSet<>());
154                         effectiveRoles.get(y).addAll(
155                             x.getModes().stream()
156                                         .map(URI::toString)
157                                         .collect(Collectors.toList()));
158                     });
159             });
160 
161         LOGGER.debug("Unfiltered ACL: {}", effectiveRoles);
162 
163         // Transform the effectiveRoles from a Set to a List.
164         return effectiveRoles.entrySet().stream()
165             .map(x -> new AbstractMap.SimpleEntry<>(x.getKey(), new ArrayList<>(x.getValue())))
166             .collect(Collectors.toMap(Map.Entry::getKey, Map.Entry::getValue));
167     }
168 
169     /**
170      *  This is a function for generating a Predicate that filters WebACAuthorizations according
171      *  to whether the given acl:accessToClass values contain any of the rdf:type values provided
172      *  when creating the predicate.
173      */
174     private Function<List<String>, Predicate<WebACAuthorization>> accessToClass = uris -> x -> {
175         return uris.stream()
176                    .distinct()
177                    .filter(y -> x.getAccessToClassURIs().contains(y))
178                    .findFirst()
179                    .isPresent();
180     };
181 
182     /**
183      *  This is a function for generating a Predicate that filters WebACAuthorizations according
184      *  to whether the given acl:accessTo values contain any of the target resource values provided
185      *  when creating the predicate.
186      */
187     private Function<List<String>, Predicate<WebACAuthorization>> accessTo = uris -> x -> {
188         return uris.stream()
189                    .distinct()
190                    .filter(y -> x.getAccessToURIs().contains(y))
191                    .findFirst()
192                    .isPresent();
193     };

195     /**
196      *  A simple predicate for filtering out any non-acl triples.
197      */
198     final Predicate<Property> isAclPredicate =
199          p -> !p.isAnon() && p.getNameSpace().startsWith(WEBAC_NAMESPACE_VALUE);
200 
201     /**
202      *  This function reads a Fedora ACL resource and all of its acl:Authorization children.
203      *  The RDF from each child resource is put into a WebACAuthorization object, and the
204      *  full list is returned.
205      *
206      *  @param location the location of the ACL resource
207      *  @return a list of acl:Authorization objects
208      */
209     private List<WebACAuthorization> getAuthorizations(final String location) {
210 
211         final Session internalSession = sessionFactory.getInternalSession();
212         final List<WebACAuthorization> authorizations = new ArrayList<>();
213         final IdentifierConverter<Resource, FedoraResource> translator =
214                 new DefaultIdentifierTranslator(internalSession);
215         final Model model = createDefaultModel();
216 
217         LOGGER.debug("Effective ACL: {}", location);
218 
219         // Find the specified ACL resource
220 
221         if (location.startsWith(FEDORA_INTERNAL_PREFIX)) {
222 
223             final FedoraResource resource = nodeService.find(internalSession,
224                     location.substring(FEDORA_INTERNAL_PREFIX.length()));
225 
226             // Read each child resource, filtering on acl:Authorization type, keeping only acl-prefixed triples.
227             resource.getChildren().forEachRemaining(child -> {
228                 if (child.getTypes().contains(WEBAC_AUTHORIZATION)) {
229                     final Map<String, List<String>> aclTriples = new HashMap<>();
230                     child.getTriples(translator, PropertiesRdfContext.class)
231                          .filter(p -> isAclPredicate.test(model.asStatement(p).getPredicate()))
232                          .forEachRemaining(t -> {
233                             aclTriples.putIfAbsent(t.getPredicate().getURI(), new ArrayList<>());
234                              if (t.getObject().isURI()) {
235                                 aclTriples.get(t.getPredicate().getURI()).add(t.getObject().getURI());
236                              } else if (t.getObject().isLiteral()) {
237                                 aclTriples.get(t.getPredicate().getURI()).add(
238                                     t.getObject().getLiteralValue().toString());
239                              }
240                          });
241                     // Create a WebACAuthorization object from the provided triples.
242                     LOGGER.debug("Adding acl:Authorization from {}", child.getPath());
243                     authorizations.add(createAuthorizationFromMap(aclTriples));
244                 }
245             });
246         }
247         return authorizations;
248     }
249 
250     private static WebACAuthorization createAuthorizationFromMap(final Map<String, List<String>> data) {
251         return new WebACAuthorization(
252                     data.getOrDefault(WEBAC_AGENT_VALUE, EMPTY),
253                     data.getOrDefault(WEBAC_AGENT_CLASS_VALUE, EMPTY),
254                     data.getOrDefault(WEBAC_MODE_VALUE, EMPTY).stream()
255                                 .map(URI::create).collect(Collectors.toList()),
256                     data.getOrDefault(WEBAC_ACCESSTO_VALUE, EMPTY),
257                     data.getOrDefault(WEBAC_ACCESSTO_CLASS_VALUE, EMPTY));
258     }

260     /**
261      * Recursively find the effective ACL as a URI along with the FedoraResource that points to it.
262      * This way, if the effective ACL is pointed to from a parent resource, the child will inherit
263      * any permissions that correspond to access to that parent. This ACL resource may or may not exist,
264      * and it may be external to the fedora repository.
265      */
266     private static Optional<Pair<URI, FedoraResource>> getEffectiveAcl(final FedoraResource resource) {
267         try {
268             final IdentifierConverter<Resource, FedoraResource> translator =
269                 new DefaultIdentifierTranslator(resource.getNode().getSession());
270             final List<String> acls = new ArrayList<>();
271             final Model model = createDefaultModel();
272 
273             resource.getTriples(translator, PropertiesRdfContext.class)
274                 .filter(t -> model.asStatement(t).getPredicate().hasURI(WEBAC_ACCESS_CONTROL_VALUE))
275                 .filter(t -> t.getObject().isURI())
276                 .forEachRemaining(t -> {
277                     acls.add(t.getObject().getURI());
278                 });
279             if (!acls.isEmpty()) {
280                 if (acls.size() > 1) {
281                     LOGGER.warn("Found multiple ACLs defined for this node. Using: {}", acls.get(0));
282                 }
283                 return Optional.of(Pair.of(URI.create(acls.get(0)), resource));
284             } else if (resource.getNode().getDepth() == 0) {
285                 LOGGER.debug("No ACLs defined on this node or in parent hierarchy");
286                 return Optional.empty();
287             } else {
288                 LOGGER.trace("Checking parent resource for ACL. No ACL found at {}", resource.getPath());
289                 return getEffectiveAcl(resource.getContainer());
290             }
291         } catch (final RepositoryException ex) {
292             LOGGER.debug("Exception finding effective ACL: {}", ex.getMessage());
293             return Optional.empty();
294         }
295     }
