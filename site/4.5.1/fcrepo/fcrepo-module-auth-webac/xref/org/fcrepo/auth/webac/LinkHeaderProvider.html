46  /**
47   * Insert WebAC Link headers to responses
48   *
49   * @author whikloj
50   * @since 2015-10-30
51   */
52  @Component
53  public class LinkHeaderProvider implements UriAwareHttpHeaderFactory {
54  
55      private static final Logger LOGGER = getLogger(LinkHeaderProvider.class);
56  
57      @Autowired
58      private SessionFactory sessionFactory;
59  
60      @Autowired
61      private NodeService nodeService;
62  
63      @Override
64      public Multimap<String, String> createHttpHeadersForResource(final UriInfo uriInfo, final FedoraResource resource) {
65  
66          final Session internalSession = sessionFactory.getInternalSession();
67          final IdentifierConverter<Resource, FedoraResource> translator =
68                  new DefaultIdentifierTranslator(internalSession);
69          final ListMultimap<String, String> headers = ArrayListMultimap.create();
70  
71          LOGGER.debug("Adding WebAC Link Header for Resource: {}", resource);
72          // Get the correct Acl for this resource
73          WebACRolesProvider.getEffectiveAcl(resource).ifPresent(acls -> {
74              // If the Acl is present we need to use the internal session to get its URI
75              nodeService.find(internalSession, acls.resource.getPath())
76              .getTriples(translator, PROPERTIES)
77              .collect(toModel()).listObjectsOfProperty(createProperty(WEBAC_ACCESS_CONTROL_VALUE))
78              .forEachRemaining(linkObj -> {
79                  if (linkObj.isURIResource()) {
80                      final Resource acl = linkObj.asResource();
81                      final String aclPath = translator.convert(acl).getPath();
82                      final URI aclUri = uriInfo.getBaseUriBuilder().path(aclPath).build();
83                      headers.put("Link", Link.fromUri(aclUri).rel("acl").build().toString());
84                  }
85              });
86          });
87  
88          return headers;
89      }

91  
92  }
